// ─────────────────────────────────────────
// Queue — Core FIFO Data Structure
// ─────────────────────────────────────────

class Queue {
  constructor() {
    this.items = [];       // internal array storage
  }

  /** Add item to the back of the queue */
  enqueue(item) {
    this.items.push(item);
  }

  /** Remove and return item from the front */
  dequeue() {
    if (this.isEmpty()) return null;
    return this.items.shift();
  }

  /** Inspect front item without removing */
  peek() {
    if (this.isEmpty()) return null;
    return this.items[0];
  }

  /** Returns true if queue has no items */
  isEmpty() { return this.items.length === 0; }
  size()    { return this.items.length; }
  toArray() { return [...this.items]; }
}


// ─────────────────────────────────────────
// PrinterQueue — Printer System Layer
// ─────────────────────────────────────────

class PrinterQueue {
  constructor() {
    this.queue     = new Queue();
    this.processed = 0;
  }

  /** Create a job object and enqueue it */
  addJob(name, pages, priority = 'normal') {
    const job = {
      id       : Date.now() + Math.random(),
      name     : name.trim() || 'Untitled',
      pages    : parseInt(pages) || 1,
      priority,
    };
    this.queue.enqueue(job);
    return job;
  }

  /** Process (print) the front job */
  processJob() {
    const job = this.queue.dequeue();
    if (job) this.processed++;
    return job;
  }

  /** Print all jobs currently in the queue */
  printQueue() { return this.queue.toArray(); }

  peek()    { return this.queue.peek(); }
  isEmpty() { return this.queue.isEmpty(); }
  clear()   { this.queue = new Queue(); }

  get count()      { return this.queue.size(); }
  get totalPages() {
    return this.queue.toArray().reduce((s, j) => s + j.pages, 0);
  }
}
